// --- ส่วนที่ 1: ปรับปรุงการ Filter และการเรียงลำดับ (บรรทัดที่ 124 เดิม) --- const filteredAssets = useMemo(() => { const term = searchTerm.toString().trim().toLowerCase(); if (!term) return []; // ค้นหาเฉพาะรายการที่ "ลงท้ายด้วย" (Ends With) const matches = allAssets.filter(asset => String(asset.AssetID || "").toLowerCase().endsWith(term) || String(asset.NOID || "").toLowerCase().endsWith(term) ); // เรียงลำดับจาก "ท้ายตารางย้อนขึ้นมาหน้า" (ข้อมูลใหม่ขึ้นก่อน) return [...matches].reverse().slice(0, 30); }, [searchTerm, allAssets]); // --- ส่วนที่ 2: ปรับปรุงฟังก์ชัน Scanner (บรรทัดที่ 150 เดิม) --- const startScanner = () => { setScannerOpen(true); setTimeout(async () => { try { const html5QrCode = new Html5Qrcode("reader"); scannerRef.current = html5QrCode; await html5QrCode.start( { facingMode: "environment" }, { fps: 20, qrbox: { width: 250, height: 250 } }, (decodedText) => { let rawId = decodedText.trim(); let targetId = rawId; // รองรับ QR หลากหลาย: ถ้าเป็น URL ให้ดึงรหัสท้ายสุดออกมา if (rawId.includes('/')) { targetId = rawId.split('/').filter(p => p !== "").pop(); } // ถ้ามีอักขระพิเศษแบ่ง ให้เอาส่วนท้ายสุด else if (rawId.includes(':') || rawId.includes('-')) { targetId = rawId.split(/[:|-]/).pop(); } stopScanner(); // ส่งค่าไปที่ช่องค้นหา เพื่อให้แสดงรายการให้เลือก (ไม่เข้า Detail ทันที) setSearchTerm(targetId); setView('search'); setShowSearch(true); } ); } catch (err) { setAlertMsg('กรุณาอนุญาตการเข้าถึงกล้อง'); setScannerOpen(false); } }, 300); };
